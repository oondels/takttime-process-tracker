instalar python 3.12:
sudo apt update
sudo apt install software-properties-common -y
sudo add-apt-repository ppa:deadsnakes/ppa -y

sudo apt update
sudo apt install -y build-essential curl git \
  libssl-dev zlib1g-dev libbz2-dev libreadline-dev libsqlite3-dev \
  libncursesw5-dev xz-utils tk-dev libffi-dev liblzma-dev

curl https://pyenv.run | bash
echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.bashrc
echo 'export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.bashrc
echo 'eval "$(pyenv init --path)"' >> ~/.bashrc
echo 'eval "$(pyenv init -)"' >> ~/.bashrc
exec $SHELL

pyenv install 3.12.7
pyenv global 3.12.7   # ou 'pyenv local 3.12.7' no diretório do projeto
python -V
python -m venv .venv && source .venv/bin/activate

colocar aplicação para rodar como servico e iniciar com sistema:
/opt/dass_apps/takttime-process-tracker/.venv/bin/python /opt/dass_apps/takttime-process-tracker/app.py
sudo nano ~/config/autostart/takttime-process-tracker.desktop
